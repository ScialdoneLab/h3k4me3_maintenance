accessibility_stages <- c("Pre-ZGA" = accessibility_peaks)
promoters_1kb <- promoters(allgenes, upstream = 1000, downstream = 1000)
stages = names(accessibility_stages)
names(stages) <- stages

overlaps <- lapply(accessibility_stages, function(x) {
  findOverlaps(promoters_1kb, x, select = "first")
})
## Warning in .merge_two_Seqinfo_objects(x, y): Each of the 2 combined objects has sequence levels not in the other:
##   - in 'x': Sca37, Sca9
##   - in 'y': Sca10, Sca14, Sca17, Sca18, Sca20, Sca26, Sca28, Sca3, Sca30, Sca6, Sca7, Sca8
##   Make sure to always combine/compare objects based on the same reference
##   genome (use suppressWarnings() to suppress this warning).
peak_widths <- lapply(stages, function(y) {
  sapply(overlaps[[y]], function(x) {
    if (is.na(x)) {
      0
    } else {
      width(accessibility_stages[[y]][x])
    }
  })
})

peak_binary <- lapply(peak_widths, function(x) {
  x > 0
})

build_df <- function(norm_matrix, binarized, widths) {
  df <- data.frame(
    "promoter.enrich" = enriched_score(norm_matrix),
    "peak" = binarized,
    "peak.width" = widths,
    "promoter.max" = apply(norm_matrix, 1, max),
    "promoter.mean" = rowMeans(norm_matrix),
    row.names = allgenes$gene_id
  )
}

accessibility_dfs <- lapply(stages, function(x) {
  build_df(accessibility_norm_mats[[x]], peak_binary[[x]], peak_widths[[x]])
})
names(accessibility_dfs) <- stages
# Save data
save(accessibility_norm_mats, accessibility_dfs, accessibility_stages, file = paste0("data/processed/accessibility.RData"))
